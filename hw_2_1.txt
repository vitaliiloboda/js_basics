Почему код дает именно такие результаты?
var a = 1, b = 1, c, d;
c = ++a; alert(c);           // 2 сначала а увеличивается на 1 и далее это число присвается с, с = 1 + 1 = 2
d = b++; alert(d);           // 1 в данном случае b увеличается на 1, но возвращает значение, которое было до увеличения - 1
c = (2+ ++a); alert(c);      // 5 сейчас а = 2 и далее а увеличивается на 1 и прибаляется к 2, d = 2 + 3 = 5
d = (2+ b++); alert(d);      // 4 сейчас b = 2, далее b увеличивается на 1, но возвращает значение, которое было до увеличения - 2, d = 2 + 2 = 4
alert(a);                    // 3 после всех операций выше а было увеличено дважны на 1
alert(b);                    // 3 после всех операций выше b было увеличено дважны на 1
